  cmake_minimum_required(VERSION 3.21)
  project(fysiomon VERSION 1.0.0 LANGUAGES CXX)

# C++ 17 is required

  set(CMAKE_CXX_STANDARD 17)
  set(CMAKE_CXX_STANDARD_REQUIRED ON)

  # install folder

set(ROOT_DIR  "/Users/jan/Library/CloudStorage/OneDrive-Personal/my projects/2024 physiomon/c-sources")
set(INSTALL_DIR "/Users/jan/")

# QT packages

  find_package(Qt6 REQUIRED COMPONENTS SerialPort Widgets)
  qt_standard_project_setup()

# application sources

  set(APPLICATION_SOURCES
    ${ROOT_DIR}/physiomon.cpp
    ${ROOT_DIR}/mainwindow.cpp
    ${ROOT_DIR}/statusbar/statusbar.cpp
    ${ROOT_DIR}/channels/channels.cpp
    ${ROOT_DIR}/displays/displays.cpp
    ${ROOT_DIR}/settings/settings.cpp
    ${ROOT_DIR}/arduinocomm/arduinocomm.cpp 
    ${ROOT_DIR}/measuredevice/measuredevice.cpp
    ${ROOT_DIR}/devphysiodaq/devphysioqaq.cpp
  )

# include folders

  include_directories(
    ${ROOT_DIR}
    ${ROOT_DIR}/statusbar
    ${ROOT_DIR}/channels
    ${ROOT_DIR}/settings
    ${ROOT_DIR}/displays
    ${ROOT_DIR}/arduinocomm/
    ${ROOT_DIR}/measuredevice
    ${ROOT_DIR}/devphysiodaq
    ${Qt6Widgets_INCLUDE_DIRS}
    ${Qt6SerialPort_INCLUDE_DIRS}
  )

# and build and install to target folder. Link with QT

  qt_add_executable(fysiomon ${APPLICATION_SOURCES})
  target_link_libraries(fysiomon PRIVATE Qt6::SerialPort Qt6::Widgets)
  set_target_properties(fysiomon PROPERTIES WIN32_EXECUTABLE ON MACOSX_BUNDLE ON)

  install(TARGETS fysiomon DESTINATION ${INSTALL_DIR})
